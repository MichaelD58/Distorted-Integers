32def DistortedRootsOfOne(n, alpha): 

33 """ 

34 Returns a list containing all the distortedint values for all elements of x in the set that satisfies the 

35 condition x*x = 1. 

36 """ 

37 if (n == 1): 

38 return [DistortedInt(0, n, 0)] 

39 # 1 modulo 1 and 0 modulo 1 are equivalent, so in the exceptional case that n == 1 

40 # x ⊗ x = 1 is equivalent to x ⊗ x = 0, which is always true for the only possible value of x and alpha which is 0 

41 # therefore, we can return a list of length 1 containing DistortedInt(0,1,0) 

42 roots = [] 

43 for i in range(0, n): 

44 x = DistortedInt(i, n, alpha) 

45 if x * x == DistortedInt(1, n, alpha): 

46 roots.append(x) 

47 return roots
